

function Get-Directory {
    param(
        $Path = $PWD.Path 
    )
    $Item = Get-Item -Path $Path

    switch ( $Item.GetType().Name ) {
        FileInfo { $DirPath = $Item.FullName }
        DirectoryInfo {$DirPath = $Item.Parent}
    }
    $Output = Get-Item -Path $DirPath
    Write-Output -InputObject $Output
}
